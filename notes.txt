## SETTING UP VIRTUAL ENV

# Create the virtual env:
virtualenv -p python3.5 .env

# Activate the virtual env:
source .env/bin/activate

# Install project-specific packages:
pip3 install -r requirements.txt

# When done working:
deactivate
